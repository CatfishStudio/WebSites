Списки

Списки - это последовательности произвольных объектов.


пустой список

       names = [];

       name = list();


получить данные списка

names = ["Dave", "Mark", "Ann", "Phil"];

a = names[2];

print(a);                                # результат Ann

изменить данные списка

names[0] = "Jeff";

b = names[0];

print(b);                                # результат Jeff

добавление нового элемента в конец списка

       names.append("John");


вставить элемент в середину списка

       names.insert(2, "Thomas");


извлечение и изменение целого фрагмента

       b = names[0:2];

       c = names[2:];

       names[1] = ‘Jeff’;

       names[0:2] = [‘Dave’, ’Mark’, ’Jeff’];


конкатенация списков

       a = [1,2,3]+[4,5];


вложенные списки

       a = [1, “Dave”, 3.14, [“Mark”, 7, 9, [100, 101]], 10];

a[1];                                # результат Dave

       a[3][2];                                # результат 9

       a[3][3][1];                        # результат 101
